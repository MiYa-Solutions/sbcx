def create_customer
  if provider
    self.customer = self.provider.customers.new(name:         new_customer,
                                                address1:     address1,
                                                address2:     address2,
                                                country:      country,
                                                city:         city,
                                                state:        state,
                                                zip:          zip,
                                                phone:        phone,
                                                mobile_phone: mobile_phone) if new_customer.present? && customer.nil?

  else
    self.customer = self.organization.customers.new(name:         new_customer,
                                                    address1:     address1,
                                                    address2:     address2,
                                                    country:      country,
                                                    city:         city,
                                                    state:        state,
                                                    zip:          zip,
                                                    phone:        phone,
                                                    mobile_phone: mobile_phone) if new_customer.present? && customer.nil?
  end
end
